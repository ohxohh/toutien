<script>

function save() {

    const s = parseInt(document.getElementById('s').value),

          d = parseInt(document.getElementById('d').value),

          w = document.getElementById('w').value,

          t = document.getElementById('t').value,

          n = document.getElementById('n').value;


    if(!s || !d || !w) return alert('數據填完整才行！');


    const dateStr = (new Date().getMonth()+1)+'/'+new Date().getDate();

    let logs = JSON.parse(localStorage.getItem('p_battle_final') || '[]');

    if(logs.find(l => l.date===dateStr && l.t===t)) return alert('今天這個時段已紀錄過囉！');


    const entry = { date: dateStr, t, w, s, d, n, ts: new Date().getTime() };

    logs.unshift(entry);

    localStorage.setItem('p_battle_final', JSON.stringify(logs));


    document.getElementById('s').value='';

    document.getElementById('d').value='';

    document.getElementById('w').value='';

    document.getElementById('n').value='';

    render();

}


function render() {

    const logs = JSON.parse(localStorage.getItem('p_battle_final') || '[]');

    const container = document.getElementById('list');

    const groups = {};

    logs.forEach(l => {

        if(!groups[l.date]) groups[l.date] = [];

        groups[l.date].push(l);

    });


    container.innerHTML = Object.keys(groups).map(date=>`

        <div class="border-2 border-[#cfc4b2] rounded-3xl overflow-hidden shadow-sm bg-[#f0ede7]">

            <div class="bg-[#e2ddcf] px-3 py-1 font-black text-gray-600 text-xs tracking-widest border-b">

                📅 日期：${date}

            </div>

            <div class="divide-y divide-[#cfc4b2]">

                ${groups[date].sort((a,b)=>a.ts-b.ts).map(l=>`

                    <div class="p-2 grid grid-cols-12 items-center gap-1">

                        <div class="col-span-3 text-sm ${l.t==='早上'?'morning':l.t==='晚上'?'evening':'migraine'}">${l.t}</div>

                        <div class="col-span-3 text-sm font-medium text-gray-700">${l.w}kg</div>

                        <div class="col-span-3 text-sm ${l.s >= 115 || l.d >= 90 ? 'high-bp' : 'text-gray-700'}">${l.s}/${l.d}</div>

                        <div class="col-span-2 text-sm text-gray-400 italic">${l.n||''}</div>

                        <div class="col-span-1 flex justify-center items-center">

                          <span class="delete-btn" onclick="deleteEntry(${l.ts})">刪除</span>

                        </div>

                    </div>

                `).join('')}

            </div>

        </div>

    `).join('');


    drawCharts(logs);

}


function deleteEntry(ts){

    if(!confirm('你確定要刪除此筆紀錄嗎？請 double check！')) return;

    let logs = JSON.parse(localStorage.getItem('p_battle_final') || '[]');

    logs = logs.filter(l=>l.ts!==ts);

    localStorage.setItem('p_battle_final', JSON.stringify(logs));

    render();

}


// Chart.js 繪圖

let weightChart, bpChart;

function drawCharts(logs){

    const dates = logs.map(l => l.date).reverse();

    const weightData = logs.map(l => parseFloat(l.w)).reverse();

    const sData = logs.map(l => l.s).reverse();

    const dData = logs.map(l => l.d).reverse();


    const weightCtx = document.getElementById('weightChart').getContext('2d');

    const bpCtx = document.getElementById('bpChart').getContext('2d');


    if(weightChart) weightChart.destroy();

    if(bpChart) bpChart.destroy();


    weightChart = new Chart(weightCtx, {

        type:'line',

        data:{ labels: dates, datasets:[{label:'體重(kg)', data:weightData, borderColor:'#7a6a57', backgroundColor:'#7a6a5733', fill:true, tension:0.3}] },

        options:{ scales:{y:{beginAtZero:false}} }

    });


    bpChart = new Chart(bpCtx, {

        type:'line',

        data:{

            labels: dates,

            datasets:[

                {label:'收縮壓', data:sData, borderColor:'#3b4f6b', backgroundColor:'#3b4f6b33', tension:0.3},

                {label:'舒張壓', data:dData, borderColor:'#d38b3b', backgroundColor:'#d38b3b33', tension:0.3}

            ]

        },

        options:{ scales:{y:{beginAtZero:false}} }

    });

}


render();
